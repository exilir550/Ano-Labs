<script src="https://cdn.jsdelivr.net/npm/ethers@5.7.2/dist/ethers.umd.min.js"></script>
<script>
  async function checkNFTOwnership() {
    const nftContract = "PASTE_YOUR_CONTRACT_ADDRESS"; // üîç Replace with your NFT contract
    const abi = [{
      "inputs": [{"internalType":"address","name":"owner","type":"address"}],
      "name": "balanceOf",
      "outputs": [{"internalType":"uint256","name":"","type":"uint256"}],
      "stateMutability": "view",
      "type": "function"
    }];

    if (!window.ethereum) return alert("Please install MetaMask.");

    const provider = new ethers.providers.Web3Provider(window.ethereum);
    await provider.send("eth_requestAccounts", []);
    const signer = provider.getSigner();
    const user = await signer.getAddress();

    const contract = new ethers.Contract(nftContract, abi, provider);
    const balance = await contract.balanceOf(user);

    if (balance.gt(0)) {
      document.getElementById("protectedContent").style.display = "block";
    } else {
      document.body.innerHTML = "
‚ùå Access Denied: NFT required
";
    }
  }
  window.addEventListener("load", checkNFTOwnership);
</script>
